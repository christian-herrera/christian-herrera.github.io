4)  Configuro para que utilize las claves GPG
      eval $(/usr/bin/ssh-pageant -r -a "/tmp/.ssh-pageant-$USERNAME")
      export SSH_AUTH_SOCK=$(gpgconf --list-dirs agent-ssh-socket)
